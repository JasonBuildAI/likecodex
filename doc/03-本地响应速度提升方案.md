## 一、当前响应速度瓶颈分析

### 1.1 一次典型交互的完整链路

```
用户输入
  ↓
1. CLI 接收输入                                      ~1ms
2. HTTP POST 到 Python 引擎 /chat                    ~2ms (本地环回)
3. 上下文组装（CacheFirstContext.build_for_llm()）   ~5ms
4. DeepSeek API 调用                                 200ms~3000ms
5. SSE 流式响应解析                                   ~1ms/token
6. 工具调用结果处理                                   ~10ms~500ms
7. 循环回到步骤 3                                    ...
  ↓
用户看到输出
```

### 1.2 瓶颈分析

| 阶段 | 耗时 | 是否可优化 | 优化目标 |
|------|------|-----------|----------|
| 进程启动 | 3-5 秒（Python + Rust 双进程） | ✅ | <0.5 秒 |
| LLM 首 token 延迟 | 200-500ms | 部分（模型侧为主） | 通过缓存预热优化 |
| 上下文构建 | 5-50ms | ✅ | <5ms |
| 同一会话第二轮 | 无冷启动 | ✅ | 维持 |
| 工具执行（读） | 5-500ms | ✅ | 缓存复用 |
| 链式串行工具 | 累加延迟 | ✅ | 并行化 |

**架构精简（方向一）已解决**：去掉 Rust 服务器跳转，节省 2-5ms/请求。

## 二、改造方案

### 2.1 后台 Shell 管理（P0）

当前模型执行 `run_command("npm run dev")` 会阻塞整个 agent 循环，直到命令完成。后台 Shell 允许长时间任务在后台运行，agent 可以同时做其他事。

**新建文件**: `likecodex_engine/tools/shell_manager.py`

```python
"""后台 Shell 会话管理器"""
import asyncio
import time
from dataclasses import dataclass, field

@dataclass
class ShellSession:
    cmd: str
    process: asyncio.subprocess.Process
    stdout_lines: list[str] = field(default_factory=list)
    stderr_lines: list[str] = field(default_factory=list)
    created_at: float = field(default_factory=time.time)
    last_read_at: float = field(default_factory=time.time)
    is_running: bool = True
    
    @property
    def output(self) -> str:
        return "".join(self.stdout_lines + self.stderr_lines)
    
    @property
    def age_seconds(self) -> float:
        return time.time() - self.created_at

class ShellManager:
    """管理多个后台 shell 会话"""
    
    def __init__(self, max_shells: int = 5):
        self._shells: dict[str, ShellSession] = {}
        self._max_shells = max_shells
        self._lock = asyncio.Lock()
    
    async def create(self, cmd: str, shell_id: str | None = None) -> str:
        """创建新的后台 shell 会话"""
        async with self._lock:
            if len(self._shells) >= self._max_shells:
                # 自动清理最旧的已完成会话
                self._cleanup_finished()
            
            sid = shell_id or f"shell_{int(time.time())}"
            process = await asyncio.create_subprocess_shell(
                cmd,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
            )
            
            session = ShellSession(cmd=cmd, process=process)
            self._shells[sid] = session
            
            # 后台读取输出
            asyncio.create_task(self._read_output(sid, session))
            
            return sid
    
    async def _read_output(self, sid: str, session: ShellSession):
        """后台持续读取进程输出"""
        try:
            async for line in session.process.stdout:
                session.stdout_lines.append(line.decode().rstrip() + "\n")
            session.is_running = False
        except Exception:
            session.is_running = False
    
    async def get_output(self, shell_id: str, tail: int = 50) -> dict:
        """获取 shell 输出（可选只取最后 N 行）"""
        session = self._shells.get(shell_id)
        if not session:
            return {"error": f"Shell {shell_id} not found"}
        
        lines = session.stdout_lines + session.stderr_lines
        if tail > 0:
            lines = lines[-tail:]
        
        session.last_read_at = time.time()
        return {
            "shell_id": shell_id,
            "output": "".join(lines),
            "is_running": session.is_running,
            "age_seconds": session.age_seconds,
        }
    
    async def send_input(self, shell_id: str, input_text: str) -> dict:
        """发送输入到正在运行的 shell"""
        session = self._shells.get(shell_id)
        if not session:
            return {"error": f"Shell {shell_id} not found"}
        if not session.is_running:
            return {"error": "Shell already terminated"}
        
        session.process.stdin.write(input_text.encode() + b"\n")
        await session.process.stdin.drain()
        return {"success": True, "shell_id": shell_id}
    
    async def kill(self, shell_id: str) -> dict:
        """终止 shell 会话"""
        session = self._shells.get(shell_id)
        if not session:
            return {"error": f"Shell {shell_id} not found"}
        
        session.process.terminate()
        session.is_running = False
        return {"success": True, "shell_id": shell_id}
    
    def _cleanup_finished(self):
        """清理已完成的会话"""
        finished = [
            sid for sid, s in self._shells.items()
            if not s.is_running
        ]
        for sid in finished:
            del self._shells[sid]

# 全局单例
shell_manager = ShellManager()
```

```python
async def bash_output(shell_id: str, tail: int = 50) -> str:

@tool(
    name="kill_shell",
    description="终止一个后台 shell 会话",
    read_only=False,

async def kill_shell(shell_id: str) -> str:
    result = await shell_manager.kill(shell_id)
    return json.dumps(result, ensure_ascii=False)

@tool(
    name="wait_job",
    description="等待后台 shell 完成（最多 timeout 秒）",
    read_only=True,
)
async def wait_job(shell_id: str, timeout: int = 30) -> str:
    session = shell_manager._shells.get(shell_id)
    if not session:
        return json.dumps({"error": "Shell not found"})
    
    try:
        await asyncio.wait_for(session.process.wait(), timeout=timeout)
        output = await shell_manager.get_output(shell_id, tail=0)
        return json.dumps(output, ensure_ascii=False)
    except asyncio.TimeoutError:
        return json.dumps({"error": "timeout", "shell_id": shell_id})
```

### 2.2 上下文构建性能优化（P0）

当前 [CacheFirstContext.build_for_llm()](file:///d:/App/AgentProjects/likecodex/likecodex/packages/likecodex-engine/likecodex_engine/context/cache_first.py) 每次调用都重新组装消息列表。对于长会话，可以缓存构建结果。

**修改文件**: `likecodex_engine/context/cache_first.py`

```python
class CacheFirstContext:
    def __init__(self, ...):
        self._cached_build: tuple[int, list[dict]] | None = None
        # _cached_build = (log_version, messages)
    
    def add_message(self, role: str, content: str, **kwargs):
        self._cached_build = None  # 使缓存失效
        self.log.append({"role": role, "content": content, **kwargs})
    
    def build_for_llm(self) -> list[dict]:
        # 如果日志版本没变，直接返回缓存
        log_version = len(self.log)
        if self._cached_build and self._cached_build[0] == log_version:
            return self._cached_build[1]
        
        # 构建完整消息列表
        messages = [self.system_prompt] + self.log
        
        # 对中间的工具调用进行配对清理
        messages = self._sanitize_tool_pairing(messages)
        
        self._cached_build = (log_version, messages)
        return messages
```

预期效果：多轮对话中，`build_for_llm()` 从 O(n) 变为 O(1)。

### 2.3 工具结果缓存（P1）

对于纯读操作，相同参数的结果可以缓存复用。

**新建文件**: `likecodex_engine/tools/cache.py`

```python
"""工具结果缓存 - 避免重复执行相同读操作"""
import hashlib
import json
import time
from collections import OrderedDict

class ToolResultCache:
    """LRU 缓存 + TTL"""
    
    def __init__(self, max_size: int = 100, default_ttl: float = 5.0):
        self._cache: OrderedDict[str, tuple[float, str]] = OrderedDict()
        self._max_size = max_size
        self._default_ttl = default_ttl
    
    def _make_key(self, tool_name: str, args: dict) -> str:
        content = json.dumps({"name": tool_name, "args": args}, sort_keys=True)
        return hashlib.md5(content.encode()).hexdigest()
    
    def get(self, tool_name: str, args: dict) -> str | None:
        key = self._make_key(tool_name, args)
        if key not in self._cache:
            return None
        
        expire_at, result = self._cache[key]
        if time.time() > expire_at:
            del self._cache[key]
            return None
        
        # LRU: 移动到末尾
        self._cache.move_to_end(key)
        return result
    
    def set(self, tool_name: str, args: dict, result: str, ttl: float | None = None):
        key = self._make_key(tool_name, args)
        expire_at = time.time() + (ttl or self._default_ttl)
        
        self._cache[key] = (expire_at, result)
        self._cache.move_to_end(key)
        
        # LRU 淘汰
        while len(self._cache) > self._max_size:
            self._cache.popitem(last=False)
    
    def invalidate(self, tool_name: str | None = None):
        """清空缓存（写操作后调用）"""
        if tool_name is None:
            self._cache.clear()
        else:
            # 只清空特定工具
            keys = [
                k for k in self._cache
                if json.loads(k).get("name") == tool_name
            ]
            for k in keys:
                del self._cache[k]

# 全局缓存实例
tool_cache = ToolResultCache()
```

**集成到工具执行器**: 修改 `likecodex_engine/agent/dispatch.py`

```python
# 在只读工具批量执行时
async def execute_readonly_tool(tool_name: str, args: dict, handler):
    # 写操作后清空缓存
    if tool_name in WRITE_TOOLS:
        tool_cache.invalidate()
    
    # 读操作尝试从缓存获取
    if tool_name in READ_TOOLS:
        cached = tool_cache.get(tool_name, args)
        if cached is not None:
            return cached
    
    result = await handler(**args)
    
    # 读操作结果缓存
    if tool_name in READ_TOOLS:
        tool_cache.set(tool_name, args, result)
    
    return result
```
