## 一、目标

在 LikeCodex IDE 中集成智能终端，核心能力：

1. **终端模拟器** — 基于 xterm.js 的全功能终端，支持多 Tab 会话

3. **错误智能诊断** — 命令执行出错时，AI 自动分析并提供修复方案
4. **命令历史搜索** — 智能搜索历史命令

### 2.2 终端管理（Rust 侧）

由于 Rust 有更好的跨平台 PTY 支持（通过 `portable-pty` crate），终端进程管理放在 Rust 层。

**新增文件**: `crates/likecodex-server/src/terminal_handler.rs`

```rust
use axum::extract::ws::{Message, WebSocket};
use portable_pty::{PtyMaster, PtySize, native_pty_system, ChildKiller};
use tokio::sync::mpsc;

pub struct TerminalSession {
    id: String,
    pty: PtyMaster,
    child: Box<dyn ChildKiller + Send>,
    tx: mpsc::Sender<String>,
}

pub struct TerminalManager {
    sessions: HashMap<String, TerminalSession>,
}

impl TerminalManager {
    /// 创建新终端会话
    pub async fn create_session(&mut self, cwd: &Path) -> String {
        let pty_system = native_pty_system();
        let pair = pty_system.openpty(PtySize {
            rows: 24, cols: 80,
            pixel_width: 0, pixel_height: 0,
        }).unwrap();
        
        let cmd = if cfg!(target_os = "windows") {
            "powershell.exe"
        } else {
            "/bin/bash"
        };
        
        let child = pair.slave.spawn_command(
            Command::new(cmd)
                .current_dir(cwd)
                .env("TERM", "xterm-256color"),
        ).unwrap();
        
        let id = format!("term_{}", uuid::Uuid::new_v4());
        let (tx, rx) = mpsc::channel::<String>(256);
        
        // 后台任务：读取 PTY 输出 → 发送到 WebSocket
        let mut reader = pair.master.try_clone_reader().unwrap();
        tokio::spawn(async move {
            let mut buf = [0u8; 4096];
            loop {
                match reader.read(&mut buf) {
                    Ok(n) if n > 0 => {
                        let _ = tx.send(String::from_utf8_lossy(&buf[..n]).to_string()).await;
                    }
                    _ => break,
                }
            }
        });
        
        let session = TerminalSession {
            id: id.clone(),
            pty: pair.master,
            child: Box::new(child),
            tx,
        };
        
        self.sessions.insert(id.clone(), session);
        id
    }
    
    /// 写入数据到终端（用户输入）
    pub fn write_to_terminal(&mut self, session_id: &str, data: &str) -> Result<(), String> {
        if let Some(session) = self.sessions.get(session_id) {
            session.pty.write_all(data.as_bytes()).map_err(|e| e.to_string())
        } else {
            Err("Session not found".to_string())
        }
    }
    
    /// 调整终端大小
    pub fn resize_terminal(&mut self, session_id: &str, rows: u16, cols: u16) {
        if let Some(session) = self.sessions.get(&session_id.to_string()) {
            session.pty.resize(PtySize { rows, cols, pixel_width: 0, pixel_height: 0 }).ok();
        }
    }
    
    /// 关闭终端会话
    pub fn kill_session(&mut self, session_id: &str) {
        if let Some(session) = self.sessions.remove(session_id) {
            session.child.kill().ok();
        }
    }
}
```

**WebSocket 路由**:

```rust
async fn terminal_ws(
    ws: WebSocketUpgrade,
    Query(params): Query<TerminalQuery>,
    State(state): State<AppState>,
) -> impl IntoResponse {
    ws.on_upgrade(move |socket| handle_terminal_socket(socket, params.id, state))
}

async fn handle_terminal_socket(socket: WebSocket, session_id: String, state: AppState) {
    let (mut sender, mut receiver) = socket.split();
    let mut terminal = state.terminal_manager.lock().await;
    
    if let Some(session) = terminal.sessions.get(&session_id) {
        let mut rx = session.tx.subscribe();
        
        // PTY → WebSocket
        tokio::spawn(async move {
            while let Ok(data) = rx.recv().await {
                if sender.send(Message::Text(data.into())).await.is_err() {
                    break;
                }
            }
        });
        
        // WebSocket → PTY
        while let Some(Ok(Message::Text(data))) = receiver.recv().await {
            terminal.write_to_terminal(&session_id, &data).ok();
        }
    }
}
```

### 2.3 前端终端组件

```tsx
// web/src/ide/terminal/TerminalPanel.tsx
import { useEffect, useRef, useState } from 'react';
import { Terminal } from 'xterm';
import { FitAddon } from 'xterm-addon-fit';
import 'xterm/css/xterm.css';

export function TerminalPanel() {
  const terminalRef = useRef<HTMLDivElement>(null);
  const [sessions, setSessions] = useState<TerminalSession[]>([]);
  const [activeSession, setActiveSession] = useState<string | null>(null);
  const [showAIInput, setShowAIInput] = useState(false);
  const termInstance = useRef<Terminal | null>(null);
  const wsRef = useRef<WebSocket | null>(null);
  
  useEffect(() => {
    if (!terminalRef.current) return;
    
    // 初始化 xterm
    const term = new Terminal({
      cursorBlink: true,
      cursorStyle: 'bar',
      fontSize: 13,
      fontFamily: "'Cascadia Code', 'Fira Code', monospace",
      theme: {
        background: '#1e1e1e',
        foreground: '#d4d4d4',
        cursor: '#d4d4d4',
        selectionBackground: '#264f78',
        black: '#000000', red: '#cd3131', green: '#0dbc79',
        yellow: '#e5e510', blue: '#2472c8', magenta: '#bc3fbc',
        cyan: '#11a8cd', white: '#e5e5e5',
      },
    });
    
    const fitAddon = new FitAddon();
    term.loadAddon(fitAddon);
    term.open(terminalRef.current);
    fitAddon.fit();
    
    termInstance.current = term;
    
    // 创建终端会话
    createTerminalSession().then(sessionId => {
      setActiveSession(sessionId);
      
      // 连接 WebSocket
      const ws = new WebSocket(`ws://localhost:8080/api/ide/terminal/ws?id=${sessionId}`);
      wsRef.current = ws;
      
      ws.onmessage = (event) => {
        term.write(event.data);
      };
      
      // 用户输入 → WebSocket
      term.onData((data) => {
        ws.send(data);
      });
    });
    
    // 窗口大小调整
    const handleResize = () => fitAddon.fit();
    window.addEventListener('resize', handleResize);
    
    return () => {
      term.dispose();
      wsRef.current?.close();
      window.removeEventListener('resize', handleResize);
    };
  }, []);
  
  // Cmd+K 激活 AI 命令输入
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k' && !e.shiftKey) {
        // 确保焦点在终端
        if (document.activeElement?.closest('.terminal-panel')) {
          e.preventDefault();
          setShowAIInput(prev => !prev);
        }
      }
    };
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, []);
  
  return (
    <div className="terminal-panel h-full flex flex-col bg-[#1e1e1e]">
      {/* Tab 栏 */}
      <div className="flex items-center h-8 bg-[#2d2d2d] px-2 gap-1">
        {sessions.map(s => (
          <div key={s.id} className="terminal-tab px-3 py-0.5 text-xs rounded cursor-pointer hover:bg-[#3d3d3d]">
            {s.name}
          </div>
        ))}
        <button className="ml-auto text-xs text-gray-400 px-2 hover:text-white"
          onClick={() => createTerminalSession()}>
          + 新终端
        </button>
      </div>
      
      {/* 终端区域 */}
      <div ref={terminalRef} className="flex-1" />
      
      {/* AI 命令输入 */}
      {showAIInput && (
        <AITerminalInput
          onCommand={(cmd) => {
            termInstance.current?.write(cmd + '\r\n');
            wsRef.current?.send(cmd + '\n');
            setShowAIInput(false);
          }}
          onClose={() => setShowAIInput(false)}
        />
      )}
    </div>
  );
}
```

### 2.4 AI 命令输入

```tsx
// web/src/ide/terminal/AITerminalInput.tsx
export function AITerminalInput({ onCommand, onClose }: AITerminalInputProps) {
  const [input, setInput] = useState('');
  const [suggestedCmd, setSuggestedCmd] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  
  const generateCommand = async (text: string) => {
    setIsLoading(true);
    try {
      const res = await fetch('/api/ide/terminal/suggest', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ description: text }),
      });
      const data = await res.json();
      setSuggestedCmd(data.command);
    } finally {
      setIsLoading(false);
    }
  };
  
  return (
    <div className="border-t border-gray-700 bg-[#252526] p-2">
      {!suggestedCmd ? (
        <textarea
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => {
            if (e.key === 'Enter' && !e.shiftKey && input.trim()) {
              e.preventDefault();
              generateCommand(input.trim());
            }
            if (e.key === 'Escape') onClose();
          }}
          placeholder="描述你想执行的命令... (Enter 生成, Esc 取消)"
          className="w-full bg-[#1e1e1e] text-white border border-gray-600 rounded px-2 py-1 text-sm resize-none"
          rows={2}
          autoFocus
        />
      ) : (
        <div>
          <div className="text-xs text-gray-400 mb-1">生成的命令:</div>
          <div className="flex items-center gap-2">
            <code className="flex-1 bg-[#1e1e1e] text-green-400 px-2 py-1 rounded text-sm font-mono">
              {suggestedCmd}
            </code>
            <button
              onClick={() => onCommand(suggestedCmd)}
              className="px-3 py-1 bg-blue-600 text-white text-xs rounded hover:bg-blue-700"
            >
              运行
            </button>
            <button
              onClick={() => setSuggestedCmd('')}
              className="px-3 py-1 bg-gray-600 text-white text-xs rounded hover:bg-gray-700"
            >
              重新描述
            </button>
            <button onClick={onClose} className="px-3 py-1 text-gray-400 text-xs hover:text-white">
              取消
            </button>
          </div>
        </div>
      )}
    </div>
  );
}
```

### 2.5 Python AI 命令生成服务

```python
# packages/likecodex-engine/likecodex_engine/terminal/ai_assistant.py

class TerminalAIAssistant:
    """终端 AI 助手"""
    
    def __init__(self, llm_provider):
        self.llm = llm_provider
    
    async def suggest_command(self, description: str, os_type: str = "windows") -> str:
        """自然语言 → Shell 命令"""
        prompt = f"""你是一个终端命令专家。将用户的自然语言描述转换为 {os_type} shell 命令。
只返回命令本身，不要解释。

用户描述: {description}

Shell 命令:"""
        
        response = await self.llm.complete(
            messages=[{"role": "user", "content": prompt}],
            model="deepseek-v4-flash",
            max_tokens=100,
            temperature=0.1,
        )
        return response.content.strip()
    
    async def diagnose_error(self, command: str, error_output: str) -> str:
        """诊断命令错误并提供修复建议"""
        prompt = f"""分析以下命令执行错误，提供修复建议。

命令: {command}
错误输出: {error_output}

请提供:
1. 错误原因（一句话）
2. 修复方案

修复建议:"""
        
        response = await self.llm.complete(
            messages=[{"role": "user", "content": prompt}],
            model="deepseek-v4-flash",
            max_tokens=300,
            temperature=0.2,
        )
        return response.content.strip()
```
