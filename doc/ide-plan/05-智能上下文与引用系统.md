## 一、目标

构建 LikeCodex 的智能上下文系统，让 AI 能像 Cursor 一样理解"用户当前在做什么"。核心能力：

1. **@ 引用系统** — 在 Chat/Composer 中输入 `@` 唤起上下文选择器，引用文件、函数、文件夹、Git 信息
2. **代码库级上下文理解** — AI 自动收集当前任务相关的文件、符号、错误信息
3. **编辑历史追踪** — 记录用户的编辑行为，帮助 AI 理解意图
4. **智能优先级排序** — 从海量上下文中选择最相关的部分传送给 LLM

### 1.1 对标 Cursor

| 能力 | Cursor | LikeCodex 目标 |
|------|--------|---------------|
| `@file` 引用文件 | ✅ | ✅ 文件树搜索 + 模糊匹配 |
| `@folder` 引用文件夹 | ✅ | ✅ 递归包含文件夹内所有文件 |
| `@function` 引用函数 | ✅ | ✅ 基于 tree-sitter 的符号索引 |
| `@git` 引用 Git 信息 | ✅ | ✅ diff/log/blame |
| `@codebase` 自动上下文 | ✅ | ⏳ Phase 2 |
| `@web` 联网搜索 | ✅ | ✅ 已有 web_search 工具（复用）|
| **自动上下文收集** | ✅ 自动包含打开的文件 | ✅ 编辑器 Tab 内容自动包含 |
| **@ 建议排序** | ✅ 按相关性智能排序 | ✅ 文件 → 函数 → 最近使用 |

### 2.2 数据模型

```typescript
// web/src/ide/context/types.ts

/** @ 引用的目标类型 */
type ContextMentionType = 
  | 'file'            // 单个文件
  | 'folder'          // 文件夹（递归包含子文件）
  | 'symbol'          // 代码符号（函数、类、变量）
  | 'git-diff'        // Git 未暂存变更
  | 'git-log'         // Git 提交历史
  | 'editor-tabs'     // 当前打开的所有 Tab
  | 'problems'        // 当前诊断错误/警告
  | 'web'             // 联网搜索
  | 'clipboard';      // 剪贴板内容

/** 上下文引用项 */
interface ContextMention {
  id: string;
  type: ContextMentionType;
  label: string;           // 显示名称
  description?: string;    // 描述（如文件路径、符号签名）
  icon: string;            // 图标标识
  /** 该引用包含的内容（实际发送给 LLM 的文本） */
  content: string;
  /** 内容的 token 估算长度 */
  tokenEstimate: number;
  /** 相关性评分（0-1），用于排序 */
  relevanceScore: number;
}

/** 最终发送给 AI 的上下文包 */
interface ContextPackage {
  /** 用户手动 @ 引用的内容 */
  mentions: ContextMention[];
  /** 自动包含的上下文 */
  autoContext: {
    /** 当前活动编辑器的文件内容 */
    activeFile?: { path: string; content: string };
    /** 当前打开的其他 Tab 文件摘要 */
    openTabs: { path: string; summary: string }[];
    /** 最近的 LSP 诊断（错误优先） */
    diagnostics: { file: string; message: string; severity: string }[];
    /** 最近的终端输出 */
    recentTerminalOutput?: string;
  };
  /** 总 token 数 */
  totalTokens: number;
}
```

### 2.3 选择器组件

```tsx
// web/src/ide/context/MentionPicker.tsx
import { useState, useRef, useEffect, useCallback } from 'react';

interface MentionPickerProps {
  /** 输入框的引用位置 */
  triggerPosition: { top: number; left: number };
  /** 搜索查询（@ 后的内容） */
  query: string;
  /** 选中引用 */
  onSelect: (mention: ContextMention) => void;
  /** 关闭选择器 */
  onClose: () => void;
}

export function MentionPicker({ triggerPosition, query, onSelect, onClose }: MentionPickerProps) {
  const [results, setResults] = useState<ContextMention[]>([]);
  const [selectedIndex, setSelectedIndex] = useState(0);
  const [isLoading, setIsLoading] = useState(false);
  const inputRef = useRef<HTMLInputElement>(null);
  
  // 搜索引用
  useEffect(() => {
    if (!query) {
      // 无查询：显示智能推荐的上下文
      loadRecommendedContext();
      return;
    }
    
    setIsLoading(true);
    const timer = setTimeout(async () => {
      const res = await fetch(`/api/ide/context/search?q=${encodeURIComponent(query)}`);
      const items = await res.json();
      setResults(items);
      setIsLoading(false);
    }, 100);  // 100ms 防抖
    
    return () => clearTimeout(timer);
  }, [query]);
  
  const handleKeyDown = (e: React.KeyboardEvent) => {
    switch (e.key) {
      case 'ArrowDown':
        e.preventDefault();
        setSelectedIndex(i => Math.min(i + 1, results.length - 1));
        break;
      case 'ArrowUp':
        e.preventDefault();
        setSelectedIndex(i => Math.max(i - 1, 0));
        break;
      case 'Enter':
        e.preventDefault();
        if (results[selectedIndex]) onSelect(results[selectedIndex]);
        break;
      case 'Escape':
        onClose();
        break;
    }
  };
  
  return (
    <div
      className="fixed z-50 bg-gray-900 border border-gray-700 rounded-lg shadow-2xl w-96"
      style={{ top: triggerPosition.top, left: triggerPosition.left }}
      onKeyDown={handleKeyDown}
    >
      {/* 搜索输入框 */}
      <div className="p-2 border-b border-gray-700">
        <input
          ref={inputRef}
          type="text"
          value={query}
          onChange={(e) => setQuery(e.target.value)}
          placeholder="搜索文件、符号..."
          className="w-full bg-gray-800 text-white border border-gray-600 rounded px-2 py-1 text-sm"
          autoFocus
        />
      </div>
      
      {/* 结果列表 */}
      <div className="max-h-80 overflow-y-auto">
        {isLoading ? (
          <div className="p-4 text-center text-gray-400 text-sm">搜索中...</div>
        ) : results.length === 0 ? (
          <div className="p-4 text-center text-gray-500 text-sm">无结果</div>
        ) : (
          results.map((item, index) => (
            <div
              key={item.id}
              className={clsx(
                'flex items-center gap-2 px-3 py-2 cursor-pointer text-sm',
                index === selectedIndex ? 'bg-blue-600 bg-opacity-30' : 'hover:bg-gray-800'
              )}
              onClick={() => onSelect(item)}
              onMouseEnter={() => setSelectedIndex(index)}
            >
              {/* 类型图标 */}
              <MentionIcon type={item.type} />
              
              {/* 名称和描述 */}
              <div className="flex-1 min-w-0">
                <div className="text-white truncate">{item.label}</div>
                {item.description && (
                  <div className="text-gray-400 text-xs truncate">{item.description}</div>
                )}
              </div>
              
              {/* Token 估算 */}
              <div className="text-gray-500 text-xs whitespace-nowrap">
                ~{item.tokenEstimate}t
              </div>
            </div>
          ))
        )}
      </div>
      
      {/* 提示栏 */}
      <div className="px-3 py-1.5 border-t border-gray-700 text-xs text-gray-500 flex gap-3">
        <span>↑↓ 导航</span>
        <span>Enter 选择</span>
        <span>Esc 关闭</span>
      </div>
    </div>
  );
}

function MentionIcon({ type }: { type: ContextMentionType }) {
  const icons: Record<ContextMentionType, string> = {
    file: '📄',
    folder: '📁',
    symbol: '🔧',
    'git-diff': '🔄',
    'git-log': '📋',
    'editor-tabs': '📑',
    problems: '⚠️',
    web: '🌐',
    clipboard: '📋',
  };

}
```

### 2.4 Python 上下文搜索 API

**新增文件**: `packages/likecodex-engine/likecodex_engine/context/mention_search.py`

```python
"""@ 引用搜索服务"""
from pathlib import Path
from typing import Optional
import re

class MentionSearchService:
    """搜索文件、符号、Git 信息等 @ 引用目标"""
    
    def __init__(self, workspace_root: Path, code_indexer=None, git_service=None):
        self.root = workspace_root
        self.indexer = code_indexer  # likecodex-indexer 的 Python 绑定
        self.git = git_service
    
    async def search(self, query: str, limit: int = 20) -> list[dict]:
        """搜索所有类型的引用"""
        results = []
        
        # 1. 搜索文件（模糊匹配）
        file_results = await self._search_files(query, limit // 2)
        results.extend(file_results)
        
        # 2. 搜索符号（函数/类/变量）
        if self.indexer:
            symbol_results = await self._search_symbols(query, limit // 2)
            results.extend(symbol_results)
        
        # 3. 特殊关键词
        if self._matches(query, ["git", "diff", "change", "修改", "变更"]):
            results.append(self._git_diff_mention())
        if self._matches(query, ["problem", "error", "diagnostic", "错误", "问题"]):
            results.append(self._problems_mention())
        if self._matches(query, ["tab", "open", "editor", "打开"]):
            results.append(self._open_tabs_mention())
        if self._matches(query, ["web", "net", "搜索", "网络"]):
            results.append(self._web_search_mention())
        
        # 4. 按相关性排序
        results.sort(key=lambda x: x["relevance_score"], reverse=True)
        
        return results[:limit]
    
    async def _search_files(self, query: str, limit: int) -> list[dict]:
        """搜索文件名和路径"""
        results = []
        query_lower = query.lower()
        
        # 使用 likecodex-indexer 的 FileIndex 进行搜索
        if self.indexer:
            matches = await self.indexer.search_files(query_lower, limit)
            for path, score in matches:
                full_path = self.root / path
                results.append({
                    "id": f"file:{path}",
                    "type": "file",
                    "label": path.name,
                    "description": str(path),
                    "icon": "📄",
                    "content": f"<file path='{path}'>\n{full_path.read_text()[:5000]}\n</file>",
                    "token_estimate": 1000,
                    "relevance_score": score,
                })
        
        # 补充：路径模糊匹配
        for path in self.root.rglob("*"):
            if len(results) >= limit:
                break
            if path.is_file() and query_lower in path.name.lower():
                rel_path = path.relative_to(self.root)
                results.append({
                    "id": f"file:{rel_path}",
                    "type": "file",
                    "label": path.name,
                    "description": str(rel_path),
                    "icon": "📄",
                    "content": f"[file: {rel_path}]\n{path.read_text()[:5000]}",
                    "token_estimate": 1000,
                    "relevance_score": 0.5,
                })
        
        return results
    
    async def _search_symbols(self, query: str, limit: int) -> list[dict]:
        """搜索代码符号"""
        if not self.indexer:
            return []
        
        matches = await self.indexer.search_symbols(query, limit)
        results = []
        for symbol in matches:
            results.append({
                "id": f"symbol:{symbol.name}",
                "type": "symbol",
                "label": symbol.name,
                "description": f"{symbol.kind} · {symbol.file_path}:{symbol.line}",
                "icon": "🔧",
                "content": f"<symbol name='{symbol.name}' file='{symbol.file_path}'>\n{symbol.content[:3000]}\n</symbol>",
                "token_estimate": 600,
                "relevance_score": symbol.score,
            })
        return results
    
    def _git_diff_mention(self) -> dict:
        return {
            "id": "context:git-diff",
            "type": "git-diff",
            "label": "Git 未提交变更",
            "description": "所有未暂存和未提交的变更",
            "icon": "🔄",
            "content": self._get_git_diff_content(),
            "token_estimate": 2000,
            "relevance_score": 0.9,
        }
    
    def _problems_mention(self) -> dict:
        return {
            "id": "context:problems",
            "type": "problems",
            "label": "问题面板",
            "description": "当前项目的所有错误和警告",
            "icon": "⚠️",
            "content": self._get_problems_content(),
            "token_estimate": 1500,
            "relevance_score": 0.7,
        }
    
    def _open_tabs_mention(self) -> dict:
        return {
            "id": "context:open-tabs",
            "type": "editor-tabs",
            "label": "所有打开的 Tab",
            "description": "编辑器中当前打开的所有文件",
            "icon": "📑",
            "content": "<open_tabs>\n...\n</open_tabs>",
            "token_estimate": 3000,
            "relevance_score": 0.8,
        }
    
    def _web_search_mention(self) -> dict:
        return {
            "id": "context:web",
            "type": "web",
            "label": "联网搜索",
            "description": "搜索互联网获取最新信息",
            "icon": "🌐",
            "content": "<web_search>启用联网搜索获取最新信息</web_search>",
            "token_estimate": 100,
            "relevance_score": 0.5,
        }
    
    def _matches(self, query: str, keywords: list[str]) -> bool:
        """检查查询是否匹配关键词"""
        query_lower = query.lower()
        return any(kw.lower() in query_lower or query_lower in kw.lower() for kw in keywords)
```

### 2.5 Chat 输入框集成

```tsx
// web/src/components/Chat.tsx 改造
// 在现有的 Chat 输入框中集成 @ 引用

export function ChatInput({ onSend }: ChatInputProps) {
  const [input, setInput] = useState('');
  const [showMentions, setShowMentions] = useState(false);
  const [mentionQuery, setMentionQuery] = useState('');
  const [cursorPosition, setCursorPosition] = useState({ top: 0, left: 0 });
  const textareaRef = useRef<HTMLTextAreaElement>(null);
  
  // 检测 @ 触发
  const handleInput = (e: React.ChangeEvent<HTMLTextAreaElement>) => {
    const value = e.target.value;
    const cursor = e.target.selectionStart;
    
    // 向前查找最近的 @
    const beforeCursor = value.slice(0, cursor);
    const atIndex = beforeCursor.lastIndexOf('@');
    
    if (atIndex !== -1 && (atIndex === 0 || beforeCursor[atIndex - 1] === ' ')) {
      const query = beforeCursor.slice(atIndex + 1);
      // 如果查询包含空格或换行，则关闭选择器
      if (!query.includes(' ') && !query.includes('\n')) {
        setShowMentions(true);
        setMentionQuery(query);
        
        // 计算浮动位置
        if (textareaRef.current) {
          const pos = getCaretPosition(textareaRef.current);
          setCursorPosition(pos);
        }
      } else {
        setShowMentions(false);
      }
    } else {
      setShowMentions(false);
    }
    
    setInput(value);
  };
  
  const handleMentionSelect = (mention: ContextMention) => {
    // 将 @查询 替换为 @引用标记
    const beforeCursor = input.slice(0, input.lastIndexOf('@'));
    const afterCursor = input.slice(input.lastIndexOf('@') + 1 + mentionQuery.length);
    const mentionTag = `@[${mention.label}](${mention.id}) `;
    
    setInput(beforeCursor + mentionTag + afterCursor);
    setShowMentions(false);
  };
  
  return (
    <div className="relative">
      <textarea
        ref={textareaRef}
        value={input}
        onChange={handleInput}
        onKeyDown={handleKeyDown}
        placeholder="输入消息，使用 @ 引用文件、符号... (Cmd+Enter 发送)"
        className="w-full bg-gray-800 text-white border border-gray-700 rounded-lg px-3 py-2 resize-none"
        rows={4}
      />
      
      {showMentions && (
        <MentionPicker
          triggerPosition={cursorPosition}
          query={mentionQuery}
          onSelect={handleMentionSelect}
          onClose={() => setShowMentions(false)}
        />
      )}
    </div>
  );
}
```

## 三、自动上下文收集

### 3.1 上下文优先级策略

当用户发送消息时，系统自动收集以下上下文（按优先级排序）：

| 优先级 | 上下文 | 说明 | Token 预算 |
|--------|--------|------|-----------|
| P0 | 用户手动 @ 引用的内容 | 最高优先级 | 用户决定 |
| P1 | 当前活动编辑器文件 | 用户正在看的内容 | 2000 tokens |
| P2 | 当前打开的其他 Tab | 用户最近打开的文件 | 1000 tokens/文件 |
| P3 | 最近的 LSP 诊断 | 当前错误/警告 | 500 tokens |
| P4 | 最近的终端输出 | 用户运行的命令 | 500 tokens |
| P5 | 代码库索引摘要 | 项目结构概览 | 1000 tokens |

### 3.2 上下文打包

```typescript
// web/src/hooks/useContextPackaging.ts
export function useContextPackaging() {
  const workspace = useWorkspaceStore();
  const tabs = useTabStore();
  
  const buildContextPackage = async (mentions: ContextMention[]): Promise<ContextPackage> => {
    const package: ContextPackage = {
      mentions,
      autoContext: {
        activeFile: undefined,
        openTabs: [],
        diagnostics: [],
      },
      totalTokens: 0,
    };
    
    // 1. 包含活动文件
    const activeTab = tabs.tabs.find(t => t.filePath === tabs.activeTabId);
    if (activeTab) {
      package.autoContext.activeFile = {
        path: activeTab.filePath,
        content: activeTab.content.slice(0, 5000),  // 最多 5000 chars
      };
    }
    
    // 2. 包含其他打开的文件摘要
    for (const tab of tabs.tabs.slice(0, 5)) {  // 最多 5 个
      if (tab.filePath !== tabs.activeTabId) {
        package.autoContext.openTabs.push({
          path: tab.filePath,
          summary: tab.content.slice(0, 1000),  // 每个文件取前 1000 chars
        });
      }
    }
    
    // 3. 获取诊断信息
    try {
      const res = await fetch('/api/ide/lsp/diagnostics/summary');
      package.autoContext.diagnostics = await res.json();
    } catch {}
    
    return package;
  };
  
  return { buildContextPackage };
}
```
