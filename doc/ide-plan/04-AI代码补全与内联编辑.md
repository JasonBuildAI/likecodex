## 一、目标

实现 Cursor 最核心的两个 AI 编辑功能：

1. **Tab 补全（AI 内联预测）** — 用户打字时，AI 预测下一步要写的代码，灰色显示，按 Tab 接受
2. **Ctrl+K 内联编辑** — 选中代码后按 Ctrl+K，用自然语言描述修改，AI 直接替换选中代码

### 1.1 对标 Cursor

| 功能 | Cursor | LikeCodex 目标 | 难度 |
|------|--------|---------------|------|
| **Tab 补全** | 多模型策略，实时预测 | ✅ 基于 DeepSeek 流式预测 | 高 |
| **Ctrl+K 编辑** | 选中 → 输入 → 实时替换 | ✅ 选中 → 聊天式输入 → Diff 展示 | 中 |
| **多行补全** | 一次补全多行 | ✅ 支持 | 高 |
| **补全位置** | 行内任意位置 | ✅ 行尾 + 空行 | 中 |
| **补全取消** | 继续打字自动取消 | ✅ 用 ghost text 自动消失 | 低 |
| **Edit 模式** | 无需选中，AI 自己决定范围 | ⏳ Phase 2 | 中 |

## 二、Tab 补全系统

### 2.1 整体架构

```
┌──────────────────────────────────────────────────────┐
│                   Monaco Editor                       │
│  ┌──────────────────────────────────────────────┐    │
│  │  编辑器内容                                     │    │
│  │  function hello() {                    │    │
│  │    console.log("Hello");         ← ghost │    │
│  │  }                            ← 灰色虚线  │    │
│  └──────────────────────────────────────────────┘    │
│                       │                               │
│  ┌────────────────────┴──────────────────────────┐   │
│  │           GhostTextManager                     │   │
│  │  - 管理幽灵文本的显示/隐藏                      │   │
│  │  - 处理 Tab 键接受补全                        │   │
│  │  - 继续打字时自动消失                          │   │
│  └────────────────────┬──────────────────────────┘   │
└───────────────────────┼──────────────────────────────┘
                        │ HTTP
┌───────────────────────┼──────────────────────────────┐
│         Python likecodex-engine                       │
│  ┌────────────────────┴──────────────────────────┐   │
│  │          InlineCompletionService               │   │
│  │  - 收集上下文（当前文件 + 附近文件 + LSP）      │   │
│  │  - 调用 DeepSeek 流式预测                      │   │
│  │  - 缓存相似位置的补全结果                       │   │
│  └─────────────────────────────────────────────┘   │
└────────────────────────────────────────────────────┘
```

### 2.2 核心数据流

```
用户输入 'c' → Monaco 触发 onType
  → GhostTextManager 收集上下文：
    - 当前文件内容（光标前后 N 行）
    - 当前文件路径和语言
    - 光标位置
    - 最近修改的文件列表
  → 发送 POST /api/ide/completion/inline
  → Python 引擎调用 DeepSeek Flash（快速模型）
  → 流式返回补全内容
  → Monaco 渲染为灰色 ghost text
  → 用户按 Tab → 接受补全
  → 用户继续打字 → ghost text 自动消失
```

### 2.3 上下文收集策略

补全质量取决于上下文质量。以下是上下文收集的优先级策略：

// web/src/hooks/useInlineCompletion.ts
interface CompletionContext {
  /** 当前文件路径 */
  filePath: string;
  /** 文件语言 */
  language: string;
  /** 光标前的内容（最多 2000 chars） */
  prefix: string;
  /** 光标后的内容（最多 500 chars） */
  suffix: string;
  /** 当前文件的导入/引用语句 */
  imports: string[];
  /** 当前函数/类的签名 */
  currentScope: string;
  /** 附近文件的导出接口（文件顶部 50 行内） */
  nearbyExports: string[];
  /** 光标位置的 LSP 信息 */
  lspInfo?: {
    /** 当前符号类型 */
    symbolKind: string;
    /** 期望的返回类型 */
    expectedType?: string;
    /** 可用变量列表 */
    variables: string[];
  };
}

function collectContext(editor: editor.IStandaloneCodeEditor): CompletionContext {
  const model = editor.getModel()!;
  const position = editor.getPosition()!;
  const filePath = getActiveFilePath();
  
  return {
    filePath,
    language: model.getLanguageId(),
    prefix: model.getValueInRange({
      startLineNumber: Math.max(1, position.lineNumber - 30),
      startColumn: 1,
      endLineNumber: position.lineNumber,
      endColumn: position.column,
    }),
    suffix: model.getValueInRange({
      startLineNumber: position.lineNumber,
      startColumn: position.column,
      endLineNumber: Math.min(model.getLineCount(), position.lineNumber + 10),
      endColumn: 1000,
    }),
    imports: extractImports(model, filePath),
    currentScope: getEnclosingScope(model, position),
    nearbyExports: [], // 从 workspaceStore 获取
    lspInfo: getLspInfo(filePath, position),
  };
}

### 2.4 Python 补全引擎

**新增文件**: `packages/likecodex-engine/likecodex_engine/completion/inline.py`

```python
"""内联补全服务 - 为 IDE 提供 AI 代码补全"""
import asyncio
import time
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class InlineCompletionRequest:
    file_path: str
    language: str
    prefix: str
    suffix: str
    imports: list[str] = field(default_factory=list)
    current_scope: str = ""
    cursor_line: int = 0
    cursor_col: int = 0

@dataclass
class InlineCompletionResult:
    text: str
    completion_id: str
    model: str
    latency_ms: int
    cache_hit: bool = False

class InlineCompletionService:
    """内联补全服务"""
    
    def __init__(self, llm_provider):
        self.llm = llm_provider  # DeepSeek Flash 模型
        self._cache: dict[str, tuple[float, str]] = {}  # 简单 LRU
        self._cache_ttl = 2.0  # 缓存 2 秒
        self._pending_tasks: dict[str, asyncio.Task] = {}
        
    async def complete(self, request: InlineCompletionRequest) -> Optional[InlineCompletionResult]:
        """生成内联补全"""
        start = time.time()
        
        # 1. 检查缓存
        cache_key = self._make_cache_key(request)
        cached = self._check_cache(cache_key)
        if cached:
            return InlineCompletionResult(
                text=cached, completion_id=cache_key,
                model="cache", latency_ms=0, cache_hit=True
            )
        
        # 2. 构建 prompt
        prompt = self._build_prompt(request)
        
        # 3. 调用 LLM（用 Flash 模型，快速）
        try:
            response = await self.llm.complete(
                messages=[{"role": "user", "content": prompt}],
                model="deepseek-v4-flash",
                max_tokens=256,  # 补全只需要少量 tokens
                temperature=0.1,  # 低温度保证确定性
                stop=["\n\n", "\n```"],  # 遇到空行或代码块结束符停止
            )
            
            result = response.content.strip()
            
            # 4. 缓存结果
            self._set_cache(cache_key, result)
            
            latency = int((time.time() - start) * 1000)
            return InlineCompletionResult(
                text=result, completion_id=cache_key,
                model="deepseek-v4-flash", latency_ms=latency
            )
        except Exception as e:
            logger.warning(f"Inline completion failed: {e}")
            return None
    
    def _build_prompt(self, request: InlineCompletionRequest) -> str:
        """构建补全 prompt"""
        # 对不同的语言使用不同的 prompt 模板
        prompt = f"""你是一个代码补全助手。根据上下文补全光标位置（<CURSOR>）的代码。
只返回补全的内容，不包含任何解释。保持缩进风格一致。

文件: {request.file_path}
语言: {request.language}

上下文：
```{request.language}
{request.prefix}<CURSOR>{request.suffix}
```

补全以下位置的代码："""
        
        if request.imports:
            prompt += f"\n导入语句：\n" + "\n".join(request.imports)
        if request.current_scope:
            prompt += f"\n当前作用域：{request.current_scope}"
            
        return prompt
    
    def _make_cache_key(self, request: InlineCompletionRequest) -> str:
        """生成缓存键（取前缀的哈希）"""
        content = f"{request.file_path}:{request.cursor_line}:{request.cursor_col}:{request.prefix[-100:]}"
        return hashlib.md5(content.encode()).hexdigest()
```

### 2.5 前端 Ghost Text 管理器

```typescript
// web/src/ide/editor/GhostTextManager.ts
import { editor, Range } from 'monaco-editor';
import { useTabStore } from './tabStore';

interface GhostTextDecoration {
  decorationId: string;
  text: string;
  startLine: number;
  startColumn: number;
}

export class GhostTextManager {
  private editor: editor.IStandaloneCodeEditor;
  private decorations: string[] = [];
  private currentGhost: GhostTextDecoration | null = null;
  private debounceTimer: number | null = null;
  private abortController: AbortController | null = null;
  
  // 配置
  private readonly DEBOUNCE_MS = 150;    // 用户停顿时才请求
  private readonly TRIGGER_CHARS = /[a-zA-Z0-9_.)\]>]/; // 触发字符
  private readonly CANCEL_DELAY = 100;   // 用户继续打字时取消延迟
  
  constructor(editor: editor.IStandaloneCodeEditor) {
    this.editor = editor;
    this.setupListeners();
  }
  
  private setupListeners() {
    // 键盘输入触发
    this.editor.onKeyUp((e) => {
      if (e.keyCode === 9) return; // Tab 键不触发
      
      const position = this.editor.getPosition()!;
      const model = this.editor.getModel()!;
      const char = model.getValueInRange({
        startLineNumber: position.lineNumber,
        startColumn: position.column - 1,
        endLineNumber: position.lineNumber,
        endColumn: position.column,
      });
      
      if (this.TRIGGER_CHARS.test(char) || char === '\n') {
        this.scheduleCompletion();
      } else {
        this.cancelGhostText();
      }
    });
    
    // Tab 键接受补全
    this.editor.onKeyDown((e) => {
      if (e.keyCode === 9 && this.currentGhost) {  // Tab
        e.preventDefault();
        this.acceptGhostText();
      }
    });
  }
  
  private scheduleCompletion() {
    if (this.debounceTimer) clearTimeout(this.debounceTimer);
    
    // 用户继续打字快速取消之前的请求
    if (this.currentGhost) {
      this.cancelGhostText();
    }
    
    this.debounceTimer = window.setTimeout(() => {
      this.requestCompletion();
    }, this.DEBOUNCE_MS);
  }
  
  private async requestCompletion() {
    const position = this.editor.getPosition()!;
    const model = this.editor.getModel()!;
    
    // 只在行尾或空行触发
    const currentLine = model.getLineContent(position.lineNumber);
    if (position.column < currentLine.length && currentLine.trim() !== '') return;
    
    // 取消之前的请求
    if (this.abortController) {
      this.abortController.abort();
    }
    this.abortController = new AbortController();
    
    try {
      const context = collectContext(this.editor);
      const res = await fetch('/api/ide/completion/inline', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(context),
        signal: this.abortController.signal,
      });
      
      const data = await res.json();
      if (data.text) {
        this.showGhostText(data.text, position);
      }
    } catch (err) {
      if (err instanceof DOMException && err.name === 'AbortError') return;
      console.warn('Completion request failed:', err);
    }
  }
  
  private showGhostText(text: string, position: editor.IPosition) {
    // 清除之前的 ghost text
    this.clearGhostText();
    
    // 创建幽灵文本的装饰器
    const range = new Range(
      position.lineNumber,
      position.column,
      position.lineNumber,
      position.column
    );
    
    this.decorations = this.editor.createDecorationsCollection([{
      range,
      options: {
        after: {
          content: text,
          inlineClassName: 'ghost-text',
        },
      },
    }]).map(d => d.id);
    
    this.currentGhost = {
      decorationId: this.decorations[0],
      text,
      startLine: position.lineNumber,
      startColumn: position.column,
    };
  }
  
  private acceptGhostText() {
    if (!this.currentGhost) return;
    
    const { text, startLine, startColumn } = this.currentGhost;
    const position = { lineNumber: startLine, column: startColumn };
    
    this.editor.executeEdits('ghost-text-accept', [{
      range: new Range(
        startLine, startColumn,
        startLine, startColumn
      ),
      text,
    }]);
    
    // 移动光标到补全末尾
    const endLine = startLine + text.split('\n').length - 1;
    const endCol = startColumn + text.split('\n').pop()!.length;
    this.editor.setPosition({ lineNumber: endLine, column: endCol });
    
    this.clearGhostText();
    
    // 通知后端接受补全（用于后续优化）
    fetch('/api/ide/completion/accepted', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        completionId: this.currentGhost?.decorationId,
        acceptedAt: Date.now(),
      }),
    }).catch(() => {});
  }
  
  private cancelGhostText() {
    if (this.currentGhost) {
      this.clearGhostText();
    }
    if (this.abortController) {
      this.abortController.abort();
      this.abortController = null;
    }
  }
  
  private clearGhostText() {
    this.editor.createDecorationsCollection(this.decorations).clear();
    this.decorations = [];
    this.currentGhost = null;
  }
  
  destroy() {
    this.clearGhostText();
    if (this.debounceTimer) clearTimeout(this.debounceTimer);
    if (this.abortController) this.abortController.abort();
  }
}
```

## 三、Ctrl+K 内联编辑

### 3.1 交互流程

```
1. 用户在编辑器中选择一段代码
2. 按 Ctrl+K → 弹出内联输入框（浮动在编辑器上）
3. 用户描述修改意图（如 "改名为 camelCase"）
4. 按 Enter 或点击确认
5. 后端生成修改后的代码
6. 弹出 Diff 视图展示变更
7. 用户点击"接受"或"拒绝"
8. 接受 → 代码直接替换；拒绝 → 恢复原样
```

### 3.2 内联输入框组件

```tsx
// web/src/ide/editor/InlineEditInput.tsx
import { useEffect, useRef, useState } from 'react';
import { editor, Range } from 'monaco-editor';

interface InlineEditInputProps {
  editor: editor.IStandaloneCodeEditor;
  selectedRange: Range;
  selectedText: string;
  onAccept: (result: string) => void;
  onCancel: () => void;
}

export function InlineEditInput({ editor, selectedRange, selectedText, onAccept, onCancel }: InlineEditInputProps) {
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [generatedCode, setGeneratedCode] = useState<string | null>(null);
  const inputRef = useRef<HTMLTextAreaElement>(null);
  
  // 计算浮动位置
  const position = editor.getScrolledVisiblePosition(selectedRange.getStartPosition());
  const editorDom = editor.getDomNode();
  const editorRect = editorDom?.getBoundingClientRect();
  
  const top = (editorRect?.top || 0) + (position?.top || 0) - 40;
  const left = (editorRect?.left || 0) + (position?.left || 0);
  
  useEffect(() => {
    inputRef.current?.focus();
  }, []);
  
  const handleSubmit = async () => {
    if (!input.trim() || isLoading) return;
    setIsLoading(true);
    
    try {
      const res = await fetch('/api/ide/edit/inline', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          filePath: useTabStore.getState().activeTabId,
          language: editor.getModel()?.getLanguageId(),
          selectedCode: selectedText,
          instruction: input.trim(),
        }),
      });
      
      const data = await res.json();
      setGeneratedCode(data.modifiedCode);
      
      // 展示 Diff
      useTabStore.getState().showDiff(
        selectedText,
        data.modifiedCode,
        'Ctrl+K 修改预览'
      );
    } catch (err) {
      console.error('Inline edit failed:', err);
    } finally {
      setIsLoading(false);
    }
  };
  
  // 如果已经生成，不再显示输入框（Diff 已弹出）
  if (generatedCode) return null;
  
  return (
    <div
      className="fixed z-50 bg-editor-background border border-gray-600 rounded-lg shadow-xl p-2"
      style={{ top, left, width: 400 }}
      onClick={(e) => e.stopPropagation()}
    >
      <div className="text-xs text-gray-400 mb-1 px-1">
        修改选中代码 ({selectedText.split('\n').length} 行):
      </div>
      <textarea
        ref={inputRef}
        value={input}
        onChange={(e) => setInput(e.target.value)}
        onKeyDown={(e) => {
          if (e.key === 'Enter' && !e.shiftKey) {
            e.preventDefault();
            handleSubmit();
          }
          if (e.key === 'Escape') onCancel();
        }}
        placeholder="描述修改... (例如: 添加参数校验、改名为 camelCase)"
        className="w-full bg-gray-800 text-white border border-gray-700 rounded px-2 py-1.5 text-sm resize-none"
        rows={2}
      />
      <div className="flex justify-between items-center mt-1 px-1">
        <span className="text-xs text-gray-500">Esc 取消 | Enter 确认</span>
        <button
          onClick={handleSubmit}
          disabled={isLoading || !input.trim()}
          className="px-3 py-1 bg-blue-600 text-white text-sm rounded hover:bg-blue-700 disabled:opacity-50"
        >
          {isLoading ? '生成中...' : '确认修改'}
        </button>
      </div>
    </div>
  );
}

### 3.3 Python 内联编辑 API

**新增文件**: `packages/likecodex-engine/likecodex_engine/completion/inline_edit.py`

```python
"""Ctrl+K 内联编辑服务"""
from pydantic import BaseModel

class InlineEditRequest(BaseModel):
    file_path: str
    language: str
    selected_code: str
    instruction: str
    surrounding_context: str = ""  # 选中代码周围的上下文

class InlineEditResponse(BaseModel):
    modified_code: str
    explanation: str = ""

class InlineEditService:
    """内联编辑服务"""
    
    def __init__(self, llm_provider):
        self.llm = llm_provider
        
    async def edit(self, request: InlineEditRequest) -> InlineEditResponse:
        """执行内联编辑"""
        
        prompt = f"""你是一个代码编辑助手。根据用户的指令修改选中的代码。
只返回修改后的代码，不包含任何解释。

文件: {request.file_path}
语言: {request.language}

选中的代码：
```{request.language}
{request.selected_code}
```

用户指令: {request.instruction}

修改后的代码（直接输出代码，不要 markdown 包裹）:"""
        
        # 使用 Pro 模型做内联编辑（需要理解语义）
        response = await self.llm.complete(
            messages=[{"role": "user", "content": prompt}],
            model="deepseek-v4-pro",
            max_tokens=4096,
            temperature=0.2,
        )
        
        modified = response.content.strip()
        # 去除可能的 markdown 代码块包裹
        modified = modified.removeprefix(f"```{request.language}").removeprefix("```")
        modified = modified.removesuffix("```").strip()
        
        return InlineEditResponse(modified_code=modified)
```

## 六、性能目标

| 指标 | 目标 | 说明 |
|------|------|------|
| 补全延迟（TP50） | < 300ms | 用户打字到看到 ghost text |
| 补全延迟（TP95） | < 800ms | 极差网络情况下 |
| 补全接受率 | > 30% | 用户实际接受的补全比例 |
| Ctrl+K 延迟 | < 1.5s | 提交到 show Diff |
| 误触发率 | < 5% | 不需要补全时出现 ghost text 的比例 |

---

## 七、验收标准

| 验收项 | 通过条件 |
|--------|----------|
| 基础补全 | 在 TypeScript/Python 文件中，输入 `function ` 后能看到 AI 补全 |
| 多行补全 | 输入后能看到完整的函数体补全 |
| Tab 接受 | 按 Tab 后补全内容正确插入编辑器 |
| Esc 取消 | 按 Esc 后 ghost text 消失 |
| 自动取消 | 继续打字后 ghost text 自动消失 |
| Ctrl+K 编辑 | 选中代码后按 Ctrl+K，输入指令能正确修改代码 |
| Diff 展示 | 修改后弹出 Diff 对比，接受/拒绝正常工作 |
