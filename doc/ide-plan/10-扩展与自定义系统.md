```typescript
// web/src/ide/settings/KeybindingEditor.tsx

interface Keybinding {
  id: string;
  command: string;
  label: string;
  keys: string[];       // 如 ["ctrl", "k"]
  when: string;         // 生效条件
}

export function KeybindingEditor() {
  const [keybindings, setKeybindings] = useState<Keybinding[]>([]);
  const [recording, setRecording] = useState<string | null>(null);
  
  const startRecording = (id: string) => {
    setRecording(id);
  };
  
  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (!recording) return;
    e.preventDefault();
    
    const keys: string[] = [];
    if (e.ctrlKey || e.metaKey) keys.push('Ctrl');
    if (e.shiftKey) keys.push('Shift');
    if (e.altKey) keys.push('Alt');
    
    if (e.key !== 'Control' && e.key !== 'Shift' && e.key !== 'Alt' && e.key !== 'Meta') {
      keys.push(e.key.toUpperCase());
      
      setKeybindings(bindings => bindings.map(b =>
        b.id === recording ? { ...b, keys } : b
      ));
      setRecording(null);
    }
  };
  
  return (
    <div tabIndex={0} onKeyDown={handleKeyDown}>
      {keybindings.map(kb => (
        <div key={kb.id} className="flex items-center justify-between py-1">
          <span className="text-sm">{kb.label}</span>
          <div className="flex items-center gap-1">
            {recording === kb.id ? (
              <span className="text-xs text-yellow-400">按下快捷键...</span>
            ) : (
              <div className="flex gap-0.5">
                {kb.keys.map((key, i) => (
                  <kbd key={i} className="px-1.5 py-0.5 bg-gray-700 text-gray-200 text-xs rounded">
                    {key}
                  </kbd>
                ))}
              </div>
            )}
            <button
              onClick={() => startRecording(kb.id)}
              className="ml-2 px-2 py-0.5 text-xs bg-gray-700 rounded hover:bg-gray-600"
            >
              {recording === kb.id ? '取消' : '修改'}
            </button>
          </div>
        </div>
      ))}
    </div>
  );
}
```

### 4.2 主题切换

```typescript
// web/src/hooks/useTheme.ts
import { useEffect, useState } from 'react';

export function useTheme() {
  const [theme, setTheme] = useState<'dark' | 'light'>(() => {
    return (localStorage.getItem('likecodex-theme') as 'dark' | 'light') || 'dark';
  });
  
  useEffect(() => {
    document.documentElement.setAttribute('data-theme', theme);
    localStorage.setItem('likecodex-theme', theme);
    
    // 更新 Monaco Editor 主题
    if (theme === 'dark') {
      monaco.editor.setTheme('vs-dark');
    } else {
      monaco.editor.setTheme('vs');
    }
  }, [theme]);
  
  return { theme, setTheme };
}
```

### 5.2 扩展加载器

```typescript
// web/src/ide/extensions/extensionLoader.ts

class ExtensionLoader {
  private extensions: Map<string, ExtensionManifest> = new Map();
  
  async loadExtensions() {
    // 扫描 .likecodex/extensions/ 目录
    const res = await fetch('/api/ide/extensions/list');
    const manifests: ExtensionManifest[] = await res.json();
    
    for (const manifest of manifests) {
      await this.loadExtension(manifest);
    }
  }
  
  async loadExtension(manifest: ExtensionManifest) {
    try {
      // 加载扩展的 JS 文件
      const module = await import(`/extensions/${manifest.id}/${manifest.main}`);
      
      // 注册命令
      if (manifest.contributes?.commands) {
        for (const cmd of manifest.contributes.commands) {
          this.registerCommand(cmd, module);
        }
      }
      
      // 注册快捷键
      if (manifest.contributes?.keybindings) {
        for (const kb of manifest.contributes.keybindings) {
          this.registerKeybinding(kb);
        }
      }
      
      this.extensions.set(manifest.id, manifest);
      console.log(`[Extensions] Loaded: ${manifest.name} v${manifest.version}`);
    } catch (err) {
      console.error(`[Extensions] Failed to load ${manifest.id}:`, err);
    }
  }
  
  private registerCommand(cmd: CommandDefinition, module: any) {
    const handler = module.activate?.()?.[cmd.id];
    if (handler) {
      CommandRegistry.register(cmd.id, handler);
    }
  }
}
```

## 六、命令面板

```tsx
// web/src/ide/CommandPalette.tsx
export function CommandPalette() {
  const [query, setQuery] = useState('');
  const [commands, setCommands] = useState<CommandDefinition[]>([]);
  
  useEffect(() => {
    // 从所有已注册的扩展收集命令
    const allCommands = CommandRegistry.getAllCommands();
    const filtered = allCommands.filter(cmd =>
      cmd.label.toLowerCase().includes(query.toLowerCase()) ||
      cmd.id.toLowerCase().includes(query.toLowerCase())
    );
    setCommands(filtered);
  }, [query]);
  
  return (
    <div className="fixed inset-0 z-50 flex items-start justify-center pt-[15vh]">
      <div className="w-[600px] bg-gray-900 border border-gray-700 rounded-lg shadow-2xl">
        <input
          value={query}
          onChange={(e) => setQuery(e.target.value)}
          placeholder="输入命令名称..."
          className="w-full bg-transparent text-white px-4 py-3 text-lg border-b border-gray-700 outline-none"
          autoFocus
        />
        
        <div className="max-h-96 overflow-y-auto p-2">
          {commands.map(cmd => (
            <div
              key={cmd.id}
              className="flex items-center justify-between px-3 py-2 rounded hover:bg-gray-800 cursor-pointer"
              onClick={() => CommandRegistry.execute(cmd.id)}
            >
              <span className="text-sm text-white">{cmd.label}</span>
              {cmd.keybindings && (
                <div className="flex gap-0.5">
                  {cmd.keybindings.map(k => (
                    <kbd className="px-1 py-0.5 bg-gray-700 text-xs rounded">{k}</kbd>
                  ))}
                </div>
              )}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}
```
