## 一、目标

将现有基础的 Monaco Editor 集成改造为 **多 Tab 多窗格的专业代码编辑器**，支持：

1. **多 Tab 管理** — 同时编辑多个文件，Tab 拖拽排序
2. **Diff 编辑器** — 代码变更前后对比（AI 修改预览）
3. **代码导航** — 跳转到定义、查找引用、符号搜索
4. **代码高亮与诊断** — 语法错误、警告内联显示
5. **多窗格布局** — 分屏并排编辑

### 1.1 对标 Cursor

### 2.2 Tab 管理模型

```typescript
// web/src/ide/editor/tabStore.ts
import { create } from 'zustand';

interface EditorTab {
  /** 文件路径（相对于工作区根） */
  filePath: string;
  /** Tab 标题（文件名） */
  title: string;
  /** 是否已修改未保存 */
  isDirty: boolean;
  /** 编辑器内容 */
  content: string;
  /** 原始内容（用于对比 dirty 状态） */
  originalContent: string;
  /** 语言 ID（Monaco 语言标识） */
  language: string;
  /** 光标位置 */
  cursorPosition: { lineNumber: number; column: number };
  /** 滚动位置 */
  scrollPosition: { scrollTop: number; scrollLeft: number };
}

type SplitLayout = 
  | { type: 'single'; activeTab: string }
  | { type: 'split-horizontal'; left: string; right: string; activeSide: 'left' | 'right' }
  | { type: 'split-vertical'; top: string; bottom: string; activeSide: 'top' | 'bottom' };

interface TabState {
  tabs: EditorTab[];
  activeTabId: string | null;
  splitLayout: SplitLayout;
  
  // Diff 状态
  diffVisible: boolean;
  diffOriginal: string;
  diffModified: string;
  diffTitle: string;
  
  // 操作
  openFile: (filePath: string) => Promise<void>;
  closeTab: (filePath: string) => void;
  closeOtherTabs: (filePath: string) => void;
  closeAllTabs: () => void;
  setActiveTab: (filePath: string) => void;
  reorderTabs: (fromIndex: number, toIndex: number) => void;
  updateTabContent: (filePath: string, content: string) => void;
  saveFile: (filePath: string) => Promise<void>;
  
  // Split 操作
  splitHorizontal: () => void;
  splitVertical: () => void;
  closeSplit: () => void;
  
  // Diff 操作
  showDiff: (original: string, modified: string, title: string) => void;
  closeDiff: () => void;
  acceptDiff: () => void;
}

export const useTabStore = create<TabState>((set, get) => ({
  tabs: [],
  activeTabId: null,
  splitLayout: { type: 'single', activeTab: '' },
  diffVisible: false,
  diffOriginal: '',
  diffModified: '',
  diffTitle: '',
  
  openFile: async (filePath: string) => {
    // 检查是否已在 tabs 中
    const existing = get().tabs.find(t => t.filePath === filePath);
    if (existing) {
      set({ activeTabId: filePath });
      return;
    }
    
    // 从后端读取文件
    const res = await fetch(`/api/ide/fs/read?path=${encodeURIComponent(filePath)}`);
    const { content, language } = await res.json();
    
    const newTab: EditorTab = {
      filePath,
      title: filePath.split('/').pop() || filePath,
      isDirty: false,
      content,
      originalContent: content,
      language,
      cursorPosition: { lineNumber: 1, column: 1 },
      scrollPosition: { scrollTop: 0, scrollLeft: 0 },
    };
    
    set(state => ({
      tabs: [...state.tabs, newTab],
      activeTabId: filePath,
    }));
  },
  
  saveFile: async (filePath: string) => {
    const tab = get().tabs.find(t => t.filePath === filePath);
    if (!tab) return;
    
    await fetch(`/api/ide/fs/write`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ path: filePath, content: tab.content }),
    });
    
    set(state => ({
      tabs: state.tabs.map(t => 
        t.filePath === filePath 
          ? { ...t, isDirty: false, originalContent: t.content }
          : t
      ),
    }));
  },
  
  showDiff: (original, modified, title) => {
    set({ diffVisible: true, diffOriginal: original, diffModified: modified, diffTitle: title });
  },
  
  acceptDiff: () => {
    const { diffModified, activeTabId } = get();
    if (!activeTabId) return;
    
    set(state => ({
      tabs: state.tabs.map(t => 
        t.filePath === activeTabId ? { ...t, content: diffModified, isDirty: true } : t
      ),
      diffVisible: false,
    }));
  },
}));
```

// web/src/ide/editor/TabBar.tsx
export function TabBar({ tabs, activeTabId }: TabBarProps) {
  return (
    <div className="flex items-center h-9 bg-tab-bar overflow-x-auto">
      {tabs.map((tab, index) => (
        <div
          key={tab.filePath}
          className={clsx(
            'tab-item flex items-center gap-1 px-3 h-full cursor-pointer',
            'border-r border-tab-border text-sm whitespace-nowrap',
            activeTabId === tab.filePath
              ? 'bg-editor-background text-white border-b-2 border-accent'
              : 'bg-tab-inactive text-gray-400 hover:text-gray-200'
          )}
          onClick={() => useTabStore.getState().setActiveTab(tab.filePath)}
          draggable
          onDragStart={(e) => e.dataTransfer.setData('text/plain', String(index))}
          onDrop={(e) => { /* 拖拽排序 */ }}
        >
          {/* 文件图标 */}
          <FileIcon fileName={tab.title} />
          
          {/* 文件名 */}
          <span>{tab.title}</span>
          
          {/* Dirty 指示器 */}
          {tab.isDirty && (
            <span className="w-2 h-2 rounded-full bg-yellow-400" />
          )}
          
          {/* 关闭按钮 */}
          <button
            onClick={(e) => {
              e.stopPropagation();
              useTabStore.getState().closeTab(tab.filePath);
            }}
            className="close-btn ml-1 p-0.5 hover:bg-gray-600 rounded"
          >
            <CloseIcon className="w-3 h-3" />
          </button>
        </div>
      ))}
      
      {/* 新 Tab 快速打开 */}
      <QuickOpenButton />
    </div>
  );
}

### 2.7 前端 LSP 集成

```typescript
// web/src/hooks/useLSP.ts
import { useCallback } from 'react';
import { editor } from 'monaco-editor';

export function useLSP(editorInstance: editor.IStandaloneCodeEditor | null) {
  const gotoDefinition = useCallback(async (filePath: string, line: number, col: number) => {
    const res = await fetch(
      `/api/ide/lsp/definition?file=${encodeURIComponent(filePath)}&line=${line}&col=${col}`
    );
    const locations = await res.json();
    return locations;
  }, []);
  
  const getCompletions = useCallback(async (
    filePath: string, line: number, col: number, prefix: string
  ) => {
    const res = await fetch(
      `/api/ide/lsp/completions?file=${encodeURIComponent(filePath)}&line=${line}&col=${col}&prefix=${encodeURIComponent(prefix)}`
    );
    return await res.json();
  }, []);
  
  // 将 LSP 诊断注册到 Monaco Editor 的标记（markers）
  const updateDiagnostics = useCallback(async (filePath: string) => {
    const res = await fetch(
      `/api/ide/lsp/diagnostics?file=${encodeURIComponent(filePath)}`
    );
    const diagnostics = await res.json();
    
    const model = editorInstance?.getModel();
    if (model) {
      editor.setModelMarkers(model, 'lsp', diagnostics.map((d: any) => ({
        severity: d.severity === 'error' ? 8 : d.severity === 'warning' ? 4 : 1,
        message: d.message,
        startLineNumber: d.range.start.line,
        startColumn: d.range.start.character,
        endLineNumber: d.range.end.line,
        endColumn: d.range.end.character,
      })));
    }
  }, [editorInstance]);
  
  return { gotoDefinition, getCompletions, updateDiagnostics };
}
```

## 三、实现步骤

### 3.1 Phase 1：多 Tab 编辑器（第 1 周）

1. 创建 `tabStore`（Zustand）

2. 创建 `DiffViewer` 组件（并排对比 + 接受/拒绝按钮）

### 3.3 Phase 3：LSP + 代码导航（第 3 周）

1. Rust 端实现 LSP 进程管理器（按需启动/关闭）
2. 实现 `goto_definition` / `find_references` / `hover` API

4. 实现诊断推送（LSP diagnostics → Monaco markers）
5. 实现代码补全集成（LSP completions → Monaco completion provider）
