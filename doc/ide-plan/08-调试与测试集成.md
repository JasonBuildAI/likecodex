### 2.4 断点管理

编辑器行号区域左侧点击设置断点，复用 03 的 Monaco Editor 集成：

```typescript
// web/src/ide/debug/breakpointManager.ts
import { editor } from 'monaco-editor';
import { useDebugStore } from './debugStore';

export class BreakpointManager {
  private editor: editor.IStandaloneCodeEditor;
  private decorations: string[] = [];
  
  setup(editor: editor.IStandaloneCodeEditor) {
    this.editor = editor;
    
    // 监听行号区域点击
    editor.onMouseDown((e) => {
      if (e.target.type === editor.MouseTargetType.GUTTER_GLYPH_MARGIN ||
          e.target.type === editor.MouseTargetType.GUTTER_LINE_NUMBERS) {
        const line = e.target.position!.lineNumber;
        const filePath = useTabStore.getState().activeTabId;
        
        if (!filePath) return;
        
        const store = useDebugStore.getState();
        const existing = store.breakpoints.find(
          bp => bp.filePath === filePath && bp.line === line
        );
        
        if (existing) {
          store.removeBreakpoint(existing.id);
        } else {
          store.addBreakpoint({ filePath, line, enabled: true });
        }
      }
    });
    
    // 订阅断点变更 → 更新装饰器
    useDebugStore.subscribe(
      (state) => state.breakpoints,
      (bps) => this.updateDecorations(bps)
    );
  }
  
  private updateDecorations(breakpoints: Breakpoint[]) {
    const filePath = useTabStore.getState().activeTabId;
    const bps = breakpoints.filter(bp => bp.filePath === filePath);
    
    this.editor.createDecorationsCollection(
      bps.map(bp => ({
        range: new Range(bp.line, 1, bp.line, 1),
        options: {
          isWholeLine: true,
          glyphMarginClassName: bps.enabled ? 'breakpoint-enabled' : 'breakpoint-disabled',
          glyphMarginHoverMessage: { value: `断点 - 第 ${bp.line} 行` },
        },
      }))
    );
  }
}
```

### 3.1 测试发现与运行

```typescript
// web/src/ide/debug/testRunner.ts

interface TestCase {
  id: string;
  name: string;
  filePath: string;
  line: number;
  status: 'pending' | 'running' | 'passed' | 'failed' | 'skipped';
  duration?: number;  // ms
  errorMessage?: string;
  errorStack?: string;
}

interface TestFile {
  path: string;
  framework: string;  // jest / pytest / cargo-test
  tests: TestCase[];
}

export function useTestRunner() {
  const [testFiles, setTestFiles] = useState<TestFile[]>([]);
  const [isRunning, setIsRunning] = useState(false);
  
  const discoverTests = async () => {
    const res = await fetch('/api/ide/tests/discover');
    const files = await res.json();
    setTestFiles(files);
  };
  
  const runAllTests = async () => {
    setIsRunning(true);
    const res = await fetch('/api/ide/tests/run', { method: 'POST' });
    const reader = res.body!.getReader();
    
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      // 解析 SSE 事件实时更新测试状态
      const data = JSON.parse(new TextDecoder().decode(value));
      updateTestStatus(data);
    }
    setIsRunning(false);
  };
  
  const runTestAtCursor = async (filePath: string, line: number) => {
    // 在编辑器中右键 → "运行测试" / 行号区域点击运行按钮
    const res = await fetch('/api/ide/tests/run-at', {
      method: 'POST',
      body: JSON.stringify({ filePath, line }),
    });
    return await res.json();
  };
  
  return { testFiles, isRunning, discoverTests, runAllTests, runTestAtCursor };
}
```

## 四、AI 辅助调试

```python
# packages/likecodex-engine/likecodex_engine/debug/ai_debug.py

class AIDebugAssistant:
    """AI 调试助手"""
    
    async def analyze_error(
        self, error_message: str, stack_trace: str,
        relevant_code: str, file_path: str
    ) -> dict:
        """分析错误并提供修复建议"""
        prompt = f"""分析以下代码错误并提供修复方案。

文件: {file_path}
错误信息: {error_message}
调用栈: {stack_trace[:2000]}
相关代码:
```{self._detect_language(file_path)}
{relevant_code[:3000]}
```

请提供:
1. 错误根因（一句话）
2. 修复方案（具体代码修改）
3. 预防措施

以 JSON 格式返回: {{"root_cause": "", "fix": "", "prevention": ""}}"""
        
        response = await self.llm.complete(
            messages=[{"role": "user", "content": prompt}],
            model="deepseek-v4-pro",
            max_tokens=1000,
            temperature=0.2,
            response_format={"type": "json_object"},
        )
        return json.loads(response.content)
```
