## 一、目标

在 LikeCodex IDE 中构建可视化的版本控制面板，对标 Cursor/VS Code 的 Git 集成：

1. **变更面板** — 查看所有未提交变更，按文件分组
2. **Diff 对比** — 点击文件查看变更 Diff（复用 03 DiffEditor）
3. **暂存管理** — 暂存/取消暂存/丢弃变更
4. **提交** — 可视化提交，自动生成提交信息
5. **分支管理** — 分支切换、创建、合并
6. **提交历史** — 浏览提交历史，查看提交详情

### 2.2 Rust Git 服务

基于 `git2` crate 提供高性能 Git 操作：

```rust
// crates/likecodex-server/src/git_handler.rs

pub struct GitService {
    repo: Mutex<Repository>,
}

impl GitService {
    pub fn open(path: &Path) -> Result<Self> {
        let repo = Repository::open(path)?;
        Ok(Self { repo: Mutex::new(repo) })
    }
    
    /// 获取工作区变更
    pub fn get_changes(&self) -> Result<Vec<GitChange>> {
        let repo = self.repo.lock().unwrap();
        let mut changes = Vec::new();
        
        // 未暂存的变更
        let mut status_opts = StatusOptions::new();
        status_opts.include_untracked(true)
            .recurse_untracked_dirs(true);
        
        for entry in repo.statuses(Some(&mut status_opts))?.iter() {
            let path = String::from_utf8_lossy(entry.path_bytes()).to_string();
            let status = entry.status();
            
            let change_type = if status.contains(Status::CURRENT) { continue }
                else if status.contains(Status::INDEX_NEW) { "added" }
                else if status.contains(Status::INDEX_MODIFIED) { "modified" }
                else if status.contains(Status::INDEX_DELETED) { "deleted" }
                else if status.contains(Status::WT_NEW) { "untracked" }
                else if status.contains(Status::WT_MODIFIED) { "modified" }
                else if status.contains(Status::WT_DELETED) { "deleted" }
                else { continue };
            
            let staged = status.contains(Status::INDEX_NEW) 
                || status.contains(Status::INDEX_MODIFIED)
                || status.contains(Status::INDEX_DELETED);
            
            changes.push(GitChange {
                path,
                change_type: change_type.to_string(),
                staged,
            });
        }
        
        Ok(changes)
    }
    
    /// 获取文件 Diff
    pub fn get_diff(&self, path: &str, staged: bool) -> Result<String> {
        let repo = self.repo.lock().unwrap();
        let mut diff_opts = DiffOptions::new();
        diff_opts.pathspec(path);
        
        let tree = repo.head()?.peel_to_tree().ok();
        let diff = if staged {
            repo.diff_tree_to_index(tree.as_ref(), None, Some(&mut diff_opts))
        } else {
            let index = repo.index()?;
            repo.diff_index_to_workdir(Some(&index), Some(&mut diff_opts))
        }?;
        
        // 生成 unified diff 格式文本
        let mut output = Vec::new();
        diff.print(DiffFormat::Patch, |_delta, _hunk, line| {
            let prefix = match line.origin() {
                '+' => b"+", '-' => b"-", ' ' => b" ",
                _ => b" ",
            };
            output.extend_from_slice(prefix);
            output.extend_from_slice(line.content());
            Ok(true)
        })?;
        
        Ok(String::from_utf8_lossy(&output).to_string())
    }
    
    /// 暂存文件
    pub fn stage_file(&self, path: &str) -> Result<()> {
        let repo = self.repo.lock().unwrap();
        let mut index = repo.index()?;
        index.add_path(Path::new(path))?;
        index.write()?;
        Ok(())
    }
    
    /// 取消暂存
    pub fn unstage_file(&self, path: &str) -> Result<()> {
        let repo = self.repo.lock().unwrap();
        let mut index = repo.index()?;
        index.remove_path(Path::new(path))?;
        index.write()?;
        Ok(())
    }
    
    /// 提交
    pub fn commit(&self, message: &str, author: &str, email: &str) -> Result<String> {
        let repo = self.repo.lock().unwrap();
        let sig = Signature::now(author, email)?;
        let tree_id = repo.index()?.write_tree()?;
        let tree = repo.find_tree(tree_id)?;
        let parent = repo.head()?.peel_to_commit().ok();
        
        let commit_id = if let Some(parent) = parent {
            repo.commit(Some("HEAD"), &sig, &sig, message, &tree, &[&parent])
        } else {
            repo.commit(Some("HEAD"), &sig, &sig, message, &tree, &[])
        }?;
        
        Ok(commit_id.to_string())
    }
}
```

### 2.3 变更面板 UI

```tsx
// web/src/ide/git/GitPanel.tsx
export function GitPanel() {
  const { changes, branches, currentBranch, stageFile, unstageFile, commit, discardChanges } = useGitStore();
  const [commitMessage, setCommitMessage] = useState('');
  const [selectedChange, setSelectedChange] = useState<string | null>(null);
  
  const stagedChanges = changes.filter(c => c.staged);
  const unstagedChanges = changes.filter(c => !c.staged);
  
  return (
    <div className="flex h-full">
      {/* 左侧：变更列表 */}
      <div className="w-80 border-r border-gray-700 flex flex-col">
        {/* 标题 + 分支 */}
        <div className="px-3 py-2 border-b border-gray-700">
          <div className="flex items-center justify-between">
            <span className="text-sm font-semibold text-white">
              源代码管理
            </span>
            <BranchSwitcher
              branches={branches}
              current={currentBranch}
            />
          </div>
          <div className="text-xs text-gray-400 mt-1">
            {changes.length} 个未提交变更
          </div>
        </div>
        
        {/* 暂存的变更 */}
        {stagedChanges.length > 0 && (
          <div>
            <div className="px-3 py-1 text-xs text-gray-400 font-medium">暂存的变更</div>
            {stagedChanges.map(c => (
              <ChangeItem key={c.path} change={c} onUnstage={() => unstageFile(c.path)} />
            ))}
          </div>
        )}
        
        {/* 未暂存的变更 */}
        <div className="flex-1 overflow-y-auto">
          <div className="px-3 py-1 text-xs text-gray-400 font-medium">变更</div>
          {unstagedChanges.map(c => (
            <ChangeItem
              key={c.path}
              change={c}
              onClick={() => setSelectedChange(c.path)}
              selected={selectedChange === c.path}
              onStage={() => stageFile(c.path)}
              onDiscard={() => discardChanges(c.path)}
            />
          ))}
        </div>
        
        {/* 提交区域 */}
        <div className="border-t border-gray-700 p-3">
          <textarea
            value={commitMessage}
            onChange={(e) => setCommitMessage(e.target.value)}
            placeholder="提交信息 (Ctrl+Enter 提交)"
            className="w-full bg-[#1e1e1e] text-white border border-gray-600 rounded px-2 py-1 text-sm resize-none"
            rows={3}
          />
          <div className="flex gap-2 mt-2">
            <button
              onClick={() => commit(commitMessage)}
              disabled={!commitMessage.trim() || stagedChanges.length === 0}
              className="flex-1 px-3 py-1.5 bg-green-600 text-white text-sm rounded hover:bg-green-700 disabled:opacity-50"
            >
              提交
            </button>
            <AIGenerateCommitMessage
              changes={stagedChanges}
              onGenerated={setCommitMessage}
            />
          </div>
        </div>
      </div>
      
      {/* 右侧：Diff 视图 */}
      <div className="flex-1">
        {selectedChange ? (
          <GitDiffViewer filePath={selectedChange} />
        ) : (
          <div className="flex items-center justify-center h-full text-gray-500 text-sm">
            选择一个文件查看变更
          </div>
        )}
      </div>
    </div>
  );
}
```
